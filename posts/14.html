# 【CMU15445】使用爬虫快速下载所有Slides和Notes

import requests
from bs4 import BeautifulSoup
import os

# 'https://15445.courses.cs.cmu.edu/spring2023/notes/' 下载笔记
url = 'https://15445.courses.cs.cmu.edu/spring2023/slides/' # 下载ppt

response = requests.get(url)
soup = BeautifulSoup(response.text, 'html.parser')

# 找到所有pdf文件
pdf_files = [node.get('href') for node in soup.select('a[href$=".pdf"]')] 

# 定义本地保存路径
download_folder = 'D:\国外公开课\CMU 15445 - Database\PPT'
if not os.path.exists(download_folder):
  os.mkdir(download_folder)

# 下载每个文件  
for pdf in pdf_files:
  file_url = url + pdf
  response = requests.get(file_url)
  
  if response.status_code == 200:
    local_path = os.path.join(download_folder, pdf)
    with open(local_path, 'wb') as f:
      f.write(response.content)
      
  else:
    print(f'下载{pdf}失败,状态码{response.status_code}')

print('下载完成!')
